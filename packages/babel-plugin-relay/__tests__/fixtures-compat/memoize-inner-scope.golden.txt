'use strict';

var _graphql2;

const ProfilePic = require('ProfilePic');

function SomeTopLevelView() {
  let _graphql = 'unrelated';

  return <View>
      <QueryRenderer environment={RelayStore} query={_graphql2 || (_graphql2 = {
      modern: function () {
        return require('./__generated__/ExampleQuery.graphql');
      },
      classic: function () {
        const RelayQL_GENERATED = require('react-relay/classic').QL,
              ProfilePic_user = ProfilePic.getFragment('user');

        return {
          kind: 'OperationDefinition',
          argumentDefinitions: [{
            defaultValue: null,
            kind: 'LocalArgument',
            name: 'id'
          }],
          name: 'ExampleQuery',
          operation: 'query',
          node: RelayQL_GENERATED`fragment ExampleQuery on Query {
  node(id: $id) {
    ...ProfilePic_user
  }
}`
        };
      }
    })} variables={{ id: '12345' }} />
    </View>;
}
