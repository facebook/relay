export type InlineFragment = {|
  +id: string;
  +name?: ?string;
  +message?: ?{|
    +text: ?string;
  |};
|};

export type InlineFragmentWithOverlappingFields = {|
  +hometown?: ?{|
    +id: string;
    +name: ?string;
    +message?: ?{|
      +text: ?string;
    |};
  |};
  +name?: ?string;
|};

export type InlineFragmentConditionalID = {|
  +id?: string;
  +name?: ?string;
|};

export type InlineFragmentKitchenSink = {|
  +actor: ?{|
    +id: string;
    +profilePicture: ?{|
      +uri: ?string;
      +width?: ?number;
      +height?: ?number;
    |};
    +name?: ?string;
  |};
|};
